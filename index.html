CLI Command

شما میتوانید persian flag up رو با npm CLI به راحتی نصب نمایید

npm installpersian-flag-up

یا به عنوان جایگزین میتوانید از دستور Yarn CLI استفاده نمایید

yarn add
persian-flag-up

CDN

با جایگذاری لینک زیر در script tag به راحتی از پیکیج استفاده نمایید

<script src='http://cdn.com/packages/persian-flag-up.js'></script>

اینپوت فارسی

در اینپوت کاربر نمیتواند لاتین تایپ کند

import { valid } from "persian-flag-up";

const input = document.querySelector("yourInputSelector");

input.addEventListener("keypress", (e) => {
    const {value} = e.target;

    if(!valid(value))  e.preventDefault();
});

import { price } from "persian-flag-up";

const input = document.querySelector("yourInputSelector");
const factor = document.querySelector("yourFactorSelector");

input.addEventListener("keypress", (e) => {
    const { value } = e.target;

    factor.innerHTML = price( value , {
        suffix: "ریال",
        toRial: true,
        toPersian: true,
        addCommas: true
    }) 
});
این یک متن ساده با اعدادی غیر فارسی همانند1234567890میباشد و این متن هم یک متن با حروف رایج عربی در زبان فارسی همانند"ي"میباشد
این متن هم همان متن با همان اعداد غیر فارسی میباشد که با این پیکیج تبدیل به اعداد فارسی شده اند ۱۲۳۴۵۶۷۸۹۰ و همچنین این حرف رایج عربی 'ی' نیز تبدیل به حرفی فارسی شده است

شماره همراه

شماره همراه خود را وارد نمایید

import { valid } from "persian-flag-up";

const input = document.querySelector("yourInputSelector");
const phoneValid = document.querySelector("yourSelector");

input.addEventListener("keypress", (e) => {
    const {value} = e.target;

    if(valid(value, { 
        allowEmpty: false,
        type: "phone" 
    }))  phoneValid.innerHTML = "Your phone is valid";
    else phoneValid.innerHTML = "Your phone is not valid !!!";
});
